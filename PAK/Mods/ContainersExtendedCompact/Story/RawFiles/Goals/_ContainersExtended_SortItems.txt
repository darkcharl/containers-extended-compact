// Sort Items
PROC
PROC_CE_SortItems((GUIDSTRING)_Item, (GUIDSTRING)_Inventory)
AND
NOT DB_GlobalFlag((FLAG)ContainersExtended_ForceSort_Disabled_5a5d0e15-8ed5-4556-b8ce-071c07ee825f)
AND
IsContainer((ITEM)_Item, 0) // Do not sort any container
AND
IsStoryItem(_Item, 0) // Do not sort story items
THEN
PROC_CE_SortItem((GUIDSTRING)_Item, _Inventory);

// Apply WEIGHT_REDUCED before moving if destination is an extraplanar container
PROC
PROC_CE_ApplyExtraplanarStatus((GUIDSTRING)_Item, (GUIDSTRING)_Container)
AND
IsTagged(_Container, (TAG)BAG_OF_HOLDING_8bbc5e54-e381-4526-b082-4d85fc510d15, 1)
AND
HasAppliedStatus(_Item, "WEIGHT_REDUCED", 0)
THEN
ApplyStatus(_Item, "WEIGHT_REDUCED", -1.0, 1);

// Get stack size and move full stack to container
PROC
PROC_CE_MoveItem((GUIDSTRING)_Item, (GUIDSTRING)_Container)
AND
GetStackAmount((ITEM)_Item, _ExactItemAmount, _)
THEN
PROC_CE_ApplyExtraplanarStatus(_Item, _Container);
ToInventory(_Item, _Container, _ExactItemAmount, 0, 0);
